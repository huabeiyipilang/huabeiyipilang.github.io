public class test {

//保存当前头节点
private static Node currentHeader;

//节点类
private static class Node{
int data = 0;
Node next = null;

Node(int data, Node next){
this.data = data;
this.next = next;
}
}

public static void main(String[] args) {

//新建单链表
Node node3 = new Node(3, null);
Node node2 = new Node(2, node3);
Node node1 = new Node(1, node2);

//翻转链表1
reverse1(node1);

//翻转链表2
Node tail = reverse2(node3);
//把尾结点next置空
tail.next = null;

System.out.println("finish");
}

//递归方法1：从头节点向尾结点翻转
private static void reverse1(Node node){
if(node.next == null){
//next为null，说明是翻转之前的尾结点，他的next指向上一次运算单元结束后的头节点。
node.next = currentHeader;
//递归结束，不在继续。
}else{
//保存下一个结点
Node nextNode = node.next;

//next指向上一次运算单元结束后的头节点
node.next = currentHeader;

//该节点作为新的头节点
currentHeader = node;

//将下一个结点继续递归
reverse1(nextNode);
}
}

//递归方法2：从尾节点向头结点翻转
private static Node reverse2(Node node){
if(node.next == null){
//next为null，说明是翻转之前的尾结点，他的next指向上一次运算单元结束后的头节点。
//返回该结点，作为翻转之后的尾节点。
return node;
}else{
//得到翻转之后的尾节点
Node tail = reverse2(node.next);
//尾节点的next指向当前节点
tail.next = node;
//返回该结点，作为翻转之后的尾节点。
return node;
}
}
}
